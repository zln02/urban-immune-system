## 작업 1: Kafka Consumer 신설 (`timescale_sink.py`)

### 목적
수집기(Producer)가 Kafka에 넣은 데이터를 읽어 TimescaleDB에 저장. 현재 파이프라인의 **빈 구멍**.

### 절차

```bash
git switch -c feature/lwh-kafka-consumer
mkdir -p pipeline/consumers
touch pipeline/consumers/__init__.py
```

**`pipeline/consumers/timescale_sink.py` 신규 작성** (박진영 제공 템플릿):

```python
"""Kafka → TimescaleDB sink consumer.

4개 토픽(uis.otc, uis.search, uis.wastewater, uis.weather)을 구독해
`layer_signals` 하이퍼테이블에 적재한다.
"""
import json
import logging
import os
from typing import Any, Dict

from kafka import KafkaConsumer
import psycopg2
from psycopg2.extras import execute_batch

logger = logging.getLogger(__name__)

TOPICS = ["uis.otc", "uis.search", "uis.wastewater", "uis.weather"]
BOOTSTRAP = os.getenv("KAFKA_BOOTSTRAP", "localhost:9092")
PG_DSN = os.getenv("DATABASE_URL", "postgresql://uis_user:uis_dev_placeholder_20260414@localhost:5432/urban_immune")

INSERT_SQL = """
INSERT INTO layer_signals (ts, layer, region, value, raw_payload)
VALUES (%(ts)s, %(layer)s, %(region)s, %(value)s, %(raw_payload)s)
ON CONFLICT DO NOTHING
"""


def run() -> None:
    consumer = KafkaConsumer(
        *TOPICS,
        bootstrap_servers=BOOTSTRAP,
        group_id="timescale-sink",
        value_deserializer=lambda v: json.loads(v.decode("utf-8")),
        auto_offset_reset="earliest",
        enable_auto_commit=True,
    )
    conn = psycopg2.connect(PG_DSN)
    batch: list[Dict[str, Any]] = []

    for msg in consumer:
        record = {
            "ts": msg.value.get("ts"),
            "layer": msg.topic.split(".")[-1],
            "region": msg.value.get("region", "UNKNOWN"),
            "value": msg.value.get("value"),
            "raw_payload": json.dumps(msg.value),
        }
        batch.append(record)
        if len(batch) >= 100:
            with conn.cursor() as cur:
                execute_batch(cur, INSERT_SQL, batch)
            conn.commit()
            logger.info("flushed %d records", len(batch))
            batch.clear()


if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO)
    run()
```
